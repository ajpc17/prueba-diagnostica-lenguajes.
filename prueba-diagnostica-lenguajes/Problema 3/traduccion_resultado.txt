#include <stdio.h>
#include <stdlib.h>

/* Funcion que calcula el factorial de n */
entero factorial(entero n) {
    si (n <= 0) {
        retornar 1;
    }
    entero resultado = 1;
    entero i;
    para (i = 1; i <= n; i++) {
        resultado = resultado * i;
    }
    retornar resultado;
}

/* Funcion que verifica si un numero es par */
entero es_par(entero numero) {
    si (numero % 2 == 0) {
        retornar 1;
    } sino {
        retornar 0;
    }
}

entero main() {
    entero n = 5;
    entero i;

    // Calcular factoriales del 1 al n
    para (i = 1; i <= n; i++) {
        entero f = factorial(i);
        printf("factorial(%d) = %d\n", i, f);
    }

    // Verificar pares e impares
    entero contador = 0;
    mientras (contador < n) {
        si (es_par(contador)) {
            printf("%d es par\n", contador);
        } sino {
            printf("%d es impar\n", contador);
        }
        contador++;
    }

    // Uso de switch
    entero dia = 3;
    segun (dia) {
        caso 1:
            printf("Lunes\n");
            romper;
        caso 2:
            printf("Martes\n");
            romper;
        caso 3:
            printf("Miercoles\n");
            romper;
        defecto:
            printf("Otro dia\n");
            romper;
    }

    retornar 0;
}
